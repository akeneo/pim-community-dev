import styled from 'styled-components';
import {Canvas, Meta, Story} from '@storybook/addon-docs/blocks';
import {PreviewGrid, PreviewCard, PreviewContainer, LabelContainer, SubTitle} from 'shared/PreviewGallery'
import {pimTheme} from './pim/index';

<Meta title="Guidelines/Colors" />

export const ColorContainer = styled(PreviewContainer)`
  background-color: ${props => props.color};
`;

export const colors = {
  'Brand Colors': [
    'purple',
  ],
  'System': [
    'green',
    'blue',
    'yellow',
    'red',
    'grey'
  ]
};

# Colors

Colors enhance the experience by providing visual clues and drawing attention to the main actions. Colors indicate which elements are interactive, how they relate to other elements and their degree of visibility. Important elements should stand out the most.

## Gallery
<Canvas>
  <Story name="Standard">
    {Object.keys(colors).map((section) => {
      return (
        <div key={section}>
          <h1>{section}</h1>
          {colors[section].map((color) => {
            return (
              <div key={color}>
                <SubTitle>{color}</SubTitle>
                <PreviewGrid>
                  {Object.keys(pimTheme.color)
                    .filter((colorCode) => 0 === colorCode.indexOf(color))
                    .sort((a, b) => a.substr(color.length) - b.substr(color.length))
                    .map((colorCode) => {
                      return (
                        <PreviewCard key={colorCode}>
                          <ColorContainer color={pimTheme.color[colorCode]} />
                          <LabelContainer>
                            {colorCode}
                          </LabelContainer>
                          <LabelContainer>
                            {pimTheme.color[colorCode].toUpperCase()}
                          </LabelContainer>
                        </PreviewCard>
                      )
                    })
                  }
                </PreviewGrid>
              </div>
            )
          })}
        </div>
      )
    })}
  </Story>
</Canvas>
