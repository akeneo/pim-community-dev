export const COMPONENTS_EXCLUDE = ['CardGrid', 'HighlightTitle', 'SectionTitle', 'Title', 'AnimateMessageBar'];

export const getComponentLink = componentName => {
  switch (componentName) {
    case 'Button':
    case 'IconButton':
      return `Components/Buttons/${componentName}`;
    case 'BooleanInput':
    case 'TextInput':
    case 'TagInput':
      return `Components/Inputs/${componentName}`;
    default:
      return `Components/${componentName}`;
  }
};
